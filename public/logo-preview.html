üìã React Usage Examples

// In index.html (add to <head>)
<link rel="icon" type="image/svg+xml" href="/favicon.svg" />
<link rel="apple-touch-icon" sizes="192x192" href="/logo-icon.svg" />

// In React component (Header/Navbar)
import logo from '/logo.svg'
import logoDark from '/logo-dark.svg'

// Light mode
<img src={logo} alt="IQX" height="40" />

// Dark mode (with theme)
<img src={isDarkMode ? logoDark : logo} alt="IQX" height="40" />

// Or use CSS to switch
.logo {
  content: url('/logo.svg');
}

@media (prefers-color-scheme: dark) {
  .logo {
    content: url('/logo-dark.svg');
  }
}
